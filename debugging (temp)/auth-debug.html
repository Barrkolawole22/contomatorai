🚀 Quick Backend Setup Check

If you don't have a backend running, here's what you need:

Minimal Express.js Backend:

const express = require('express');
const cors = require('cors');
const app = express();

app.use(cors());
app.use(express.json());

// Health check
app.get('/api/health', (req, res) => {
    res.json({ status: 'OK', timestamp: new Date() });
});

// Mock auth endpoints
app.post('/api/auth/register', (req, res) => {
    const { email, password, confirmPassword, name } = req.body;
    
    // Basic validation
    if (password !== confirmPassword) {
        return res.status(400).json({
            success: false,
            message: "Validation failed",
            errors: [{ field: "confirmPassword", message: "Password confirmation does not match password" }]
        });
    }
    
    res.json({ 
        token: 'mock_token_12345', 
        user: { id: 1, email, name } 
    });
});

app.post('/api/auth/login', (req, res) => {
    res.json({ 
        token: 'mock_token_12345', 
        user: { id: 1, email: req.body.email, name: 'Test User' } 
    });
});

app.get('/api/auth/profile', (req, res) => {
    const auth = req.headers.authorization;
    if (!auth || !auth.startsWith('Bearer ')) {
        return res.status(401).json({ error: 'No token provided' });
    }
    res.json({ id: 1, email: 'test@example.com', name: 'Test User' });
});

app.listen(5000, () => console.log('Server running on port 5000'));

Save as server.js, run: npm install express cors then node server.js
